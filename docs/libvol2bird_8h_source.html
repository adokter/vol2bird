201 // The data needed for calculating bird densities are collected
202 // in one big array, 'points'. Although this array is one
203 // variable, it is partitioned into 'nLayers' parts. The parts
204 // are not equal in size, therefore we need to keep track of
205 // where the data pertaining to a certain altitude bin can be
206 // written. The valid range of indexes into 'points' are stored
207 // in arrays 'indexFrom' and 'indexTo'.
208 
209 struct vol2birdPoints {
210  // the 'points' array has this many pseudo-columns
211  int nColsPoints;
212  // the 'points' array has this many rows
213  int nRowsPoints;
214  // the psuedo-column in 'points' that holds the azimuth angle
215  int azimAngleCol;
216  // the psuedo-column in 'points' that holds the elevation angle
217  int elevAngleCol;
218  // the psuedo-column in 'points' that holds the dbz value
219  int dbzValueCol;
220  // the psuedo-column in 'points' that holds the vrad value
221  int vradValueCol;
222  // the psuedo-column in 'points' that holds the cell value
223  int cellValueCol;
224  // the psuedo-column in 'points' that holds the gate classification code
225  int gateCodeCol;
226  // the psuedo-column in 'points' that holds the nyquist velocity
227  int nyquistCol;
228  // the psuedo-column in 'points' that holds the dealiased vrad value
229  int vraddValueCol;
230  // the psuedo-column in 'points' that holds the static clutter map value
231  int clutValueCol;
232  // the 'points' array itself
233  float* points; // Is allocated in vol2birdSetUp() and freed in vol2birdTearDown()
234  // for a given altitude layer in the profile, only part of the 'points'

243 };
